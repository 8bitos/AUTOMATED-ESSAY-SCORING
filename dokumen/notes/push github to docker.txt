# PUSH GITHUB -> DOCKER HUB (via GitHub Actions) + UPDATE EC2

# 0) Jalankan dari root project ini
pwd

# 1) Commit & push perubahan ke GitHub (trigger workflow)
git add .
git commit -m "update"
git push origin main

# 2) Pantau workflow di GitHub:
# Actions -> "Build and Push Docker Images"
# Pastikan status: Success

# 3) Update di EC2 (repo kamu di ~/sage hanya docker-compose)
ssh ubuntu@YOUR_EC2_PUBLIC_IP
cd ~/sage
docker compose pull backend frontend
docker compose up -d --force-recreate backend frontend
docker compose ps
docker compose logs --tail=100 backend frontend

# 4) Cek aplikasi
# Frontend: http://YOUR_EC2_PUBLIC_IP:3001
# Backend:  http://YOUR_EC2_PUBLIC_IP:8080

# NOTES:
# - GitHub repo harus punya secrets:
#   DOCKERHUB_USERNAME
#   DOCKERHUB_TOKEN
# - Workflow file: .github/workflows/docker-publish.yml
